EXEMPLOS DE SQL (LINGUAGEM DE BANCO DE DADOS)
TODOS OS EXEMPLOS SÃO SOBRE A TB_TEMA


Para inserir um registro na tabela

INSERT INTO <TABELA>
(CAMPO1, CAMPO2...)
VALUES
(VALOR1, VALOR2...)

EXEMPLO:
INSERT INTO TB_TEMA
(S_TIT_TEMA, S_DESC_TEMA)
VALUES
('BULLDOG', 'BULLDOG COM UM OSSO NA BOCA')


Para alterar um registro na tabela
UPDATE <TABELA> SET
CAMPO1 = VALOR1,
CAMPO2 = VALOR2,...
WHERE CAMPO* = VALOR*

EXEMPLO:
UPDATE TB_TEMA SET
S_DESC_TEMA = 'BULLDOG MORDENDO UM OSSO'
WHERE I_COD_TEMA = 3


Para Excluir um ou mais registros na tabela
DELETE FROM <TABELA>
WHERE CAMPO* = VALOR*

EXEMPLO:
DELETE FROM TB_TEMA
WHERE I_COD_TEMA = 2


Para Selecionar um ou mais registros na tabela
SELECT * FROM <TABELA>
WHERE CAMPO* = VALOR*
EXEMPLO:
SELECT * FROM TB_TEMA
WHERE I_COD_TEMA = 3
